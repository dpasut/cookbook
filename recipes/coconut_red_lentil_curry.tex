\begin{recipe}{$\star$Coconut Red Lentil Curry}{6 servings - 280 calories}{}

\freeform From Epicurious.

\Ing{\textbf{Marinated Tofu}}
\Ing{300g extra firm tofu}
\Ing{Lemon juice}
\Ing{Cumin or curry powder of choice\\}
\Ing{\textbf{Curry}}
\Ing{1 medium onion, finely chopped}
\Ing{2 tablespoons vegetable oil}
\Ing{1 tablespoon finely chopped peeled fresh ginger}
\Ing{2 garlic cloves, finely chopped}
\Ing{1 teaspoon ground cumin}
\Ing{\nicefrac{1}{2} teaspoon ground coriander}
\Ing{1 teaspoon turmeric}
\Ing{1 teaspoon salt}
\Ing{1 (2 \nicefrac{1}{2}-inch) fresh jalape√±o or serrano chile, finely chopped, including seeds}
\Ing{2 cups water}
\Ing{1 \nicefrac{1}{2} cups dried red lentils (10 oz)}
\Ing{1 (13- to 14-oz) can unsweetened coconut milk}
\Ing{1 lb zucchini (2 medium), cut into \nicefrac{1}{4}-inch dice}
\Ing{1 cup loosely packed fresh cilantro sprigs}

Cook onion in oil in a 3 1/2- to 4-quart heavy pot over moderate heat, stirring occasionally, until edges are golden, about 6 minutes. Add ginger and garlic and cook, stirring, 1 minute. Add cumin, coriander, turmeric, salt, and chile and cook, stirring, 1 minute.\\

Stir in water, lentils, and coconut milk, then simmer, covered, stirring occasionally, 5 minutes. Stir in zucchini and marinated tofu and simmer, covered, until lentils and zucchini are tender, about 15 minutes. Season with salt and serve on rice with cilantro sprigs scattered on top.

\end{recipe}
